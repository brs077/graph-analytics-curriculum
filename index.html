Step 1 Graph Theory Vocabulary & Mental Models ▾

Learning Objectives

  • Define nodes, edges, and graphs — and recognize them in everyday data.
  • Distinguish directed vs. undirected, weighted vs. unweighted graphs.
  • Understand paths, cycles, degree, and connected components.
  • Set up a working Python environment with Google Colab + NetworkX.

Reading & Video

Networks, Crowds, and Markets — Ch. 1 & 2. YouTube: search "graph theory introduction" from Reducible or TrevTutor for visual primers on nodes, edges, adjacency, and degree.

Key Concepts

Node (vertex): An entity — person, city, gene, web page. Edge (link): A relationship — friendship, road, interaction. Directed edge: A → B (Twitter follow). Undirected: A — B (Facebook friendship). Degree: Edge count per node; splits into in/out-degree for directed graphs. Path: Edge sequence connecting two nodes. Cycle: Path returning to its start. Connected component: Maximal set of mutually reachable nodes.

Lab 1
Environment Setup & Your First Graph
Goal: Create a Google Colab notebook, verify NetworkX, and build a tiny friendship graph.
# Cell 1 — Verify setup
import networkx as nx
import matplotlib.pyplot as plt
print(nx.__version__)

# Cell 2 — Build a small undirected friendship graph
G = nx.Graph()
G.add_nodes_from(["Alice", "Bob", "Carol", "Dave", "Eve"])
G.add_edges_from([
    ("Alice", "Bob"), ("Alice", "Carol"),
    ("Bob", "Carol"), ("Carol", "Dave"), ("Dave", "Eve"),
])

# Cell 3 — Inspect
print(f"Nodes: {G.number_of_nodes()}")
print(f"Edges: {G.number_of_edges()}")
print(f"Degree: {dict(G.degree())}")
print(f"Connected: {nx.is_connected(G)}")

# Cell 4 — Visualize
plt.figure(figsize=(6, 4))
nx.draw(G, with_labels=True, node_color="#58a6ff",
       font_color="white", node_size=800, edge_color="#555")
plt.show()

Lab 2
Directed Graphs & Weighted Edges
Goal: Model a directed, weighted flight network and explore how directionality changes analysis.
# Directed + weighted flight network
F = nx.DiGraph()
routes = [
    ("NYC","LAX",250), ("LAX","NYC",280), ("NYC","CHI",120),
    ("CHI","DEN",150), ("DEN","LAX",180), ("LAX","SFO",70),
    ("SFO","NYC",300),
]
for s, d, c in routes:
    F.add_edge(s, d, weight=c)

# In-degree vs out-degree
for n in F.nodes():
    print(f"{n}: in={F.in_degree(n)}, out={F.out_degree(n)}")

# Cheapest path NYC → SFO
path = nx.shortest_path(F, "NYC", "SFO", weight="weight")
cost = nx.shortest_path_length(F, "NYC", "SFO", weight="weight")
print(f"Cheapest: {' → '.join(path)} (${cost})")

# Visualize with edge labels
pos = nx.spring_layout(F, seed=42)
plt.figure(figsize=(8,5))
nx.draw(F, pos, with_labels=True, node_color="#d29922",
       font_color="white", node_size=900, arrows=True, arrowsize=20)
nx.draw_networkx_edge_labels(F, pos,
    edge_labels=nx.get_edge_attributes(F, "weight"), font_size=8)
plt.show()

Stretch: Add DEN→CHI for $90. Does cheapest NYC→SFO change? Use nx.find_cycle(F, source="NYC") to find cycles.

Lab 5
BFS, DFS & Dijkstra
Goal: Compare traversal orders and weighted vs. unweighted shortest paths.
G = nx.Graph()
G.add_weighted_edges_from([
    ("A","B",4),("A","C",2),("B","D",3),("B","E",1),
    ("C","D",5),("C","F",6),("D","E",2),("E","F",4),
])

bfs = ["A"] + [v for _,v in nx.bfs_edges(G,"A")]
dfs = ["A"] + [v for _,v in nx.dfs_edges(G,"A")]
print("BFS:", " → ".join(bfs))
print("DFS:", " → ".join(dfs))

dij = nx.dijkstra_path(G,"A","F")
print(f"Dijkstra A→F: {' → '.join(dij)} (cost {nx.dijkstra_path_length(G,'A','F')})")
hop = nx.shortest_path(G,"A","F")
print(f"Fewest hops: {' → '.join(hop)} ({len(hop)-1} hops)")

Key question: Why do Dijkstra and fewest-hops pick different routes? When does each matter?

Lab 7
Four Centralities on Karate Club
Goal: Compute all four centralities, rank nodes by each, and see how rankings diverge.
import networkx as nx
import matplotlib.pyplot as plt

K = nx.karate_club_graph()

centralities = {
    "Degree": nx.degree_centrality(K),
    "Betweenness": nx.betweenness_centrality(K),
    "Closeness": nx.closeness_centrality(K),
    "Eigenvector": nx.eigenvector_centrality(K),
}

# Top 5 per centrality
for name, scores in centralities.items():
    top = sorted(scores, key=scores.get, reverse=True)[:5]
    print(f"{name:14} top 5: {top}")

# Visualize each as node size
fig, axes = plt.subplots(2,2, figsize=(14,12))
pos = nx.spring_layout(K, seed=42)

for ax, (name, scores) in zip(axes.flat, centralities.items()):
    sizes = [3000 * scores[n] + 100 for n in K]
    colors = [scores[n] for n in K]
    nx.draw(K, pos, ax=ax, node_size=sizes, node_color=colors,
           cmap=plt.cm.YlGnBu, with_labels=True, font_size=7,
           edge_color="#444")
    ax.set_title(name, fontsize=13)
plt.tight_layout(); plt.show()

Explore: Node 0 tops degree centrality, but does it always top betweenness? Which nodes are "bridges" but not hubs? Why does eigenvector centrality reward connection to other well-connected nodes?

Lab 8
Finding Communities in the Karate Club
Goal: Run multiple community detection algorithms, compare results to ground-truth factions, and evaluate modularity.
import networkx as nx
from networkx.algorithms.community import (
    louvain_communities, label_propagation_communities,
    girvan_newman, modularity
)
import matplotlib.pyplot as plt

K = nx.karate_club_graph()
pos = nx.spring_layout(K, seed=42)

# Ground truth
truth = [
    {n for n in K if K.nodes[n]["club"]=="Mr. Hi"},
    {n for n in K if K.nodes[n]["club"]=="Officer"},
]

# Three algorithms
louvain   = list(louvain_communities(K, seed=42))
label_prop = list(label_propagation_communities(K))
gn_iter   = girvan_newman(K)
gn_2      = next(gn_iter)  # first split = 2 communities
gn_2      = [set(c) for c in gn_2]

results = {
    "Ground Truth": truth,
    "Louvain": louvain,
    "Label Prop": label_prop,
    "Girvan-Newman": gn_2,
}

# Modularity comparison
for name, comms in results.items():
    mod = modularity(K, comms)
    print(f"{name:16} communities={len(comms)}  modularity={mod:.4f}")

# Side-by-side visualization
fig, axes = plt.subplots(1,4, figsize=(20,5))
palette = ["#58a6ff","#d29922","#3fb950","#f47067","#bc8cff"]
for ax, (name, comms) in zip(axes, results.items()):
    color_map = {}
    for i, comm in enumerate(comms):
        for n in comm:
            color_map[n] = palette[i % len(palette)]
    colors = [color_map[n] for n in K]
    nx.draw(K, pos, ax=ax, node_color=colors, with_labels=True,
           node_size=400, font_size=7, edge_color="#444")
    ax.set_title(name)
plt.tight_layout(); plt.show()

Explore: Which algorithm best recovers the ground-truth split? What does a higher modularity score actually mean in terms of internal vs. external edges?

Lab 9
NetworkX → Gephi Pipeline
Goal: Export a graph with attributes from NetworkX, import into Gephi, and apply layout + styling.
# In Colab: export Karate Club with centrality as node attributes
import networkx as nx

K = nx.karate_club_graph()
bc = nx.betweenness_centrality(K)
comms = list(nx.algorithms.community.louvain_communities(K, seed=42))

# Attach attributes to nodes
for n in K:
    K.nodes[n]["betweenness"] = round(bc[n], 4)
    for i, c in enumerate(comms):
        if n in c:
            K.nodes[n]["community"] = i

# Export as GEXF (Gephi's native format)
nx.write_gexf(K, "karate_enriched.gexf")
print("Saved karate_enriched.gexf — download and open in Gephi")

Gephi Desktop walkthrough (or use Gephi Lite for a lighter workflow):

  • File → Open → karate_enriched.gexf
  • Layout panel → ForceAtlas2 → Run until stable → Stop
  • Appearance → Nodes → Size → Ranking → "betweenness" → Min 10, Max 60 → Apply
  • Appearance → Nodes → Color → Partition → "community" → Apply
  • Preview → Refresh → Export as PNG or SVG

Lab 12
Advanced Cypher on Movies
Goal: Write increasingly complex queries — variable-length paths, shortest paths, recommendations.
// Actors within 3 hops of Kevin Bacon
MATCH (bacon:Person {name:"Kevin Bacon"})-[:ACTED_IN*1..3]-(other:Person)
RETURN DISTINCT other.name
LIMIT 20

// Shortest path between two actors
MATCH path = shortestPath(
  (a:Person {name:"Tom Hanks"})-[*]-(b:Person {name:"Keanu Reeves"})
)
RETURN path, length(path) AS hops

// Movie recommendation: people who liked what I liked also liked...
MATCH (me:Person {name:"Tom Hanks"})-[:ACTED_IN]->(m)<-[:ACTED_IN]-(coactor)
       -[:ACTED_IN]->(rec)
WHERE NOT (me)-[:ACTED_IN]->(rec)
RETURN rec.title, count(coactor) AS score
ORDER BY score DESC LIMIT 10

// Create your own data
CREATE (you:Person {name: "Your Name"})
WITH you
MATCH (m:Movie {title: "The Matrix"})
CREATE (you)-[:REVIEWED {rating: 9, summary: "Mind-bending"}]->(m)

Lab 13
Model & Import a Custom Dataset
Goal: Take a CSV dataset, design a graph schema, import it, and query it in Neo4j.
// Example: import an airport routes CSV
// CSV columns: source_airport, dest_airport, airline, distance

// Step 1: Create constraints
CREATE CONSTRAINT airport_code IF NOT EXISTS
FOR (a:Airport) REQUIRE a.code IS UNIQUE

// Step 2: Load nodes
LOAD CSV WITH HEADERS FROM "https://your-url/routes.csv" AS row
MERGE (src:Airport {code: row.source_airport})
MERGE (dst:Airport {code: row.dest_airport})
MERGE (src)-[:ROUTE {
  airline: row.airline,
  distance: toInteger(row.distance)
}]->(dst)

// Step 3: Verify
MATCH (a:Airport)
RETURN count(a) AS airports

// Step 4: Query — busiest airports
MATCH (a:Airport)-[r:ROUTE]-()
RETURN a.code, count(r) AS routes
ORDER BY routes DESC LIMIT 10

Lab 14
Cypher Analytics Workbench
Goal: Write analytical queries on your imported dataset — degree distribution, path analysis, and simple recommendations.
// Degree distribution via Cypher
MATCH (a:Airport)
WITH a, size((a)-[:ROUTE]-()) AS degree
RETURN degree, count(a) AS frequency
ORDER BY degree

// Shortest route between two airports

// Find triangles (mutual routes via a common hub)
MATCH (a:Airport)-[:ROUTE]->(hub)-[:ROUTE]->(b:Airport)-[:ROUTE]->(a)

Lab 15
Project a Graph & Run PageRank
Goal: Create a GDS projection and run your first production algorithm.
// Create an in-memory graph projection
CALL gds.graph.project(
  'movies',
  ['Person', 'Movie'],
  {ACTED_IN: {orientation: 'UNDIRECTED'}}
)

// Run PageRank in stream mode (returns results, doesn't write)
CALL gds.pageRank.stream('movies')
YIELD nodeId, score
WITH gds.util.asNode(nodeId) AS node, score
WHERE 'Person' IN labels(node)
RETURN node.name AS person, round(score, 4) AS pagerank
ORDER BY pagerank DESC LIMIT 10

// Write PageRank back to nodes as a property
CALL gds.pageRank.write('movies', {writeProperty: 'pagerank'})
YIELD nodePropertiesWritten, ranIterations

Compare: How does GDS PageRank ranking compare to the degree centrality ranking you computed in NetworkX? Are the same people at the top?

Scalable Community Detection & Similarity
Goal: Detect communities and compute pairwise similarity on your projected graph.
// Louvain community detection
CALL gds.louvain.stream('movies')
YIELD nodeId, communityId
WITH gds.util.asNode(nodeId) AS node, communityId
RETURN communityId, count(node) AS size,
       collect(node.name)[..5] AS sample_members
ORDER BY size DESC LIMIT 10

// Node similarity — who has the most similar acting portfolio?
CALL gds.nodeSimilarity.stream('movies')
YIELD node1, node2, similarity
WITH gds.util.asNode(node1) AS p1,
     gds.util.asNode(node2) AS p2, similarity
WHERE 'Person' IN labels(p1)
RETURN p1.name, p2.name, round(similarity, 3) AS sim
ORDER BY sim DESC LIMIT 10

Lab 17
Predicting Missing Connections
Goal: Use link prediction scores to suggest relationships that probably exist but aren't in the data.
// Common neighbors score for pairs not yet connected
MATCH (p1:Person)-[:ACTED_IN]->(m)<-[:ACTED_IN]-(p2:Person)
WHERE p1 < p2
  AND NOT (p1)-[:ACTED_IN]->()<-[:ACTED_IN]-(p2)
WITH p1, p2, count(m) AS commonMovies
RETURN p1.name, p2.name, commonMovies
ORDER BY commonMovies DESC LIMIT 10

// GDS link prediction (Adamic-Adar)
MATCH (a:Person {name:"Tom Hanks"})
MATCH (b:Person)
WHERE a <> b AND NOT (a)-[:ACTED_IN]->()<-[:ACTED_IN]-(b)
RETURN b.name,
       gds.alpha.linkprediction.adamicAdar(a, b) AS score
ORDER BY score DESC LIMIT 10

Node Classification with GCN on Cora
Goal: Train a 2-layer GCN to classify paper topics in the Cora citation network.
# !pip install torch torch-geometric

import torch
import torch.nn.functional as F
from torch_geometric.datasets import Planetoid
from torch_geometric.nn import GCNConv

# Load Cora dataset
dataset = Planetoid(root='/tmp/Cora', name='Cora')
data = dataset[0]
print(f"Nodes: {data.num_nodes}, Edges: {data.num_edges}")
print(f"Features: {data.num_node_features}, Classes: {dataset.num_classes}")

# Define a simple 2-layer GCN
class GCN(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = GCNConv(dataset.num_node_features, 16)
        self.conv2 = GCNConv(16, dataset.num_classes)

    def forward(self, data):
        x = self.conv1(data.x, data.edge_index)
        x = F.relu(x)
        x = F.dropout(x, training=self.training)
        x = self.conv2(x, data.edge_index)
        return F.log_softmax(x, dim=1)

# Train
model = GCN()
optimizer = torch.optim.Adam(model.parameters(), lr=0.01, weight_decay=5e-4)

model.train()
for epoch in range(200):
    optimizer.zero_grad()
    out = model(data)
    loss = F.nll_loss(out[data.train_mask], data.y[data.train_mask])
    loss.backward()
    optimizer.step()

# Evaluate
model.eval()
pred = model(data).argmax(dim=1)
correct = (pred[data.test_mask] == data.y[data.test_mask]).sum()
acc = int(correct) / int(data.test_mask.sum())
print(f"Test accuracy: {acc:.4f}")

GNN Link Prediction on Cora
Goal: Predict missing citation links using learned node embeddings.
from torch_geometric.transforms import RandomLinkSplit
from torch_geometric.nn import GCNConv
from sklearn.metrics import roc_auc_score

# Split edges into train/val/test
transform = RandomLinkSplit(
    num_val=0.1, num_test=0.1,
    add_negative_train_samples=True
)
train_data, val_data, test_data = transform(data)

# Encoder: GCN producing node embeddings
class Encoder(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = GCNConv(data.num_node_features, 64)
        self.conv2 = GCNConv(64, 32)

    def forward(self, x, edge_index):
        x = self.conv1(x, edge_index).relu()
        return self.conv2(x, edge_index)

# Decoder: dot product between node embeddings
def decode(z, edge_index):
    return (z[edge_index[0]] * z[edge_index[1]]).sum(dim=1)

encoder = Encoder()
optimizer = torch.optim.Adam(encoder.parameters(), lr=0.01)

# Training loop (simplified)
for epoch in range(100):
    encoder.train()
    z = encoder(train_data.x, train_data.edge_index)
    pos_score = decode(z, train_data.edge_label_index[:, train_data.edge_label==1])
    neg_score = decode(z, train_data.edge_label_index[:, train_data.edge_label==0])
    loss = F.binary_cross_entropy_with_logits(
        torch.cat([pos_score, neg_score]),
        torch.cat([torch.ones_like(pos_score), torch.zeros_like(neg_score)])
    )
    optimizer.zero_grad(); loss.backward(); optimizer.step()

# Evaluate
encoder.eval()
z = encoder(test_data.x, test_data.edge_index)
scores = decode(z, test_data.edge_label_index).sigmoid()
auc = roc_auc_score(test_data.edge_label.numpy(), scores.detach().numpy())
print(f"Link prediction AUC: {auc:.4f}")

Lab 22
Graph Classification on MUTAG
Goal: Classify entire molecular graphs as mutagenic or not using a GNN + pooling architecture.
from torch_geometric.datasets import TUDataset
from torch_geometric.loader import DataLoader
from torch_geometric.nn import GCNConv, global_mean_pool

dataset = TUDataset(root='/tmp/MUTAG', name='MUTAG')
print(f"Graphs: {len(dataset)}, Classes: {dataset.num_classes}")

# Train/test split
torch.manual_seed(42)
perm = torch.randperm(len(dataset))
train_data = dataset[perm[:150]]
test_data  = dataset[perm[150:]]

class GraphClassifier(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = GCNConv(dataset.num_node_features, 64)
        self.conv2 = GCNConv(64, 32)
        self.fc = torch.nn.Linear(32, dataset.num_classes)

    def forward(self, data):
        x = self.conv1(data.x, data.edge_index).relu()
        x = self.conv2(x, data.edge_index).relu()
        x = global_mean_pool(x, data.batch)  # graph-level
        return self.fc(x)

model = GraphClassifier()
loader = DataLoader(train_data, batch_size=32, shuffle=True)
optimizer = torch.optim.Adam(model.parameters(), lr=0.01)

for epoch in range(50):
    model.train()
    for batch in loader:
        out = model(batch)
        loss = F.cross_entropy(out, batch.y)
        optimizer.zero_grad(); loss.backward(); optimizer.step()

# Test accuracy
model.eval()
test_loader = DataLoader(test_data, batch_size=64)
correct = total = 0
for batch in test_loader:
    pred = model(batch).argmax(dim=1)
    correct += (pred == batch.y).sum().item()
    total += batch.y.size(0)
print(f"Test accuracy: {correct/total:.4f}")
